def fib(N)
{
    if(N == 0 | N == 1)
    {
        return 1;
    }
    else
    {
        return fib(N - 1) + fib(N - 2);
    }
}
def pow(base, exponent)
{
    result = base;
    counter = 1;
    while counter < exponent {
        result = result * base;
        counter = counter + 1;
    }
    return result;
}
n = 15;
